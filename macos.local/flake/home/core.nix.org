#+title: Home Configuration

Set up home manager for my user account and import submodules. Define my user as =jjanzen= and specify the wallpaper and add items to the =PATH= variable.
#+begin_src nix
  { config, pkgs, ... }:

  {
    imports = [
      ./clang-format.nix
      ./fonts.nix
      ./programs/core.nix
      ./scripts/core.nix
    ];

    home.stateVersion = "24.05";
    programs.home-manager.enable = true;

    home.username = "jjanzen";
    home.homeDirectory = "/Users/jjanzen";

    home.file.".wallpaper".source = ./.wallpaper;

    home.sessionPath = [
      "$HOME/.local/bin"
      "/usr/local/bin"
      "/opt/homebrew/bin"
      "/opt/homebrew/sbin"
    ];

    home.sessionSearchVariables = {
      MANPATH = [
        "/Library/Developer/CommandLineTools/SDKs/macOSX13.0.sdk/usr/share/man"
        "/usr/share/man"
      ];
    };
  }
#+end_src
