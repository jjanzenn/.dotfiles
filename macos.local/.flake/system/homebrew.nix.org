#+title: Homebrew Configuration

Set up Homebrew.
#+begin_src nix
  { config, pkgs, ... }:

  {
    homebrew = {
      enable = true;
#+end_src

Do not quarantine casks.
#+begin_src nix
  caskArgs.no_quarantine = true;
#+end_src

Don't automatically update except with Nix update and clean up any packages that are not listed.
#+begin_src nix
  global.autoUpdate = false;
  onActivation = {
    autoUpdate = false;
    cleanup = "zap";
    upgrade = true;
  };
#+end_src
