#+title: =mpd= Configuration

Configure =mpd= as a music player.
#+begin_src nix
  { config, pkgs, ... }:

  {
    home.packages = with pkgs; [
      mpd
    ];
    home.file."./.config/mpd/mpd.conf".text = ''
      music_directory = "/Volumes/Media/Music";
      audio_output {
        type "osx"
        name "CoreAudio"
        mixer_type "software"
        enabled "yes"
      }
    '';
  }
#+end_src
