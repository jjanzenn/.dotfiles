#+title: SSH Configuration

Configure SSH. Keys should be added to the SSH agent. Specify the key file and use the system keychain.
#+begin_src nix
  { config, pkgs, ... }:
  {
    programs.ssh = {
      enable = true;
      addKeysToAgent = "yes";
      matchBlocks = {
        "*" = {
          identityFile = "~/.ssh/id_ed25519";
          extraOptions = {
            "UseKeychain" = "yes";
          };
        };
#+end_src
